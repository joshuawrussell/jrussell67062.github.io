switch (e.Action)
{
    case FileMenu_View.FileHandlerActions.Created:
    case FileMenu_View.FileHandlerActions.Opened:
        projectTree.DataContext = _fmv._ProjectTree;
        TreeControls.DataContext = _fmv._ProjectTree;
        treeScrollViewer.Visibility = Visibility.Visible;
        File_Path.Text = e.FilePath + "\\" + e.FileName;
        _sv.setProjectPath(e.FilePath + "\\" + e.FileName);
        _statusLink = e.FilePath;
        TreeControls.Visibility = Visibility.Visible;
        projectTree.Visibility = Visibility.Visible;
        Status.Text = "Project Loaded";
        _closeMenu = true;
        _sv.SetBindings(_fmv._ProjectTree);
        SelectItem();
        RefreshMetrics();
        break;
    case FileMenu_View.FileHandlerActions.Saved:
        Status.Text = "Saved: " + e.FileName + " " + DateTime.Now.ToShortTimeString();
        _closeMenu = true;
        break;
    case FileMenu_View.FileHandlerActions.SavedAs:
        _closeMenu = true;
        break;
    case FileMenu_View.FileHandlerActions.Canceled:
        _closeMenu = true;
        break;
    case FileMenu_View.FileHandlerActions.Closed:
        TreeControls.Visibility = Visibility.Collapsed;
        projectTree.Visibility = Visibility.Collapsed;
        projectTree.ContextMenu = null;
        _sv.setProjectPath(string.Empty);
        Status.Text = "Project Closed";
        _statusLink = string.Empty;
        File_Path.Text = string.Empty;
        _closeMenu = true;
        ItemContentControl.Content = null;
        break;
    case FileMenu_View.FileHandlerActions.UpdateStatus:
        Status.Text = e.StatusMessage;
        _statusLink = e.FilePath;
        break;


}
